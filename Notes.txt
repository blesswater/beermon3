# To create databases
CREATE TABLE Project (
    id INTEGER PRIMARY KEY AUTOINCREMENT, 
    proj_name TEXT,
    activity int);
# activity: 0 -> inactive (stale) not recording; 
#           1 -> active data is being collected


CREATE TABLE Probes (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    proj_id INTEGER,
    probe_name TEXT,
    probe_chan INTEGER,
    type VARCHAR(8),
    min_range FLOAT,
    max_range FLOAT);
# Types:
#    NTC_00 -> Thermistor
#    Setpoint -> Control setpoint

CREATE TABLE Temperature (
   id INTEGER PRIMARY KEY AUTOINCREMENT,
   proj_id INTEGER,
   probe_id INTEGER,
   temp_time INTEGER,
   temp FLOAT );

CREATE TABLE Event (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    proj_id INTEGER,
    event_type TEXT,
    event_time INTEGER );

# Dealing w/ timestamps
select date(temp_time, 'localtime'), 
       time(temp_time, 'localtime'), 
       datetime(temp_time, 'localtime'), 
       strftime( '%H', temp_time, 'localtime'), 
       temp 
from temperature 
where proj_id=1 and probe_id=3
